200     private int locationInGrid(Double locationInCanvas, Double canvasSize, int gridSize) {
201         double threshold = 0.1;
202         long location = Math.round(
203                 (locationInCanvas / canvasSize) * (gridSize - 1) - threshold
204         );
205 
206         return (int) location;
207     }

209     private void drawMineField(Canvas canvas, MiinaharavaLogic logic) {
210         Tile[][] mineField = logic.minefield.getMinefield();
211         int fieldX = mineField.length;
212         int fieldY = mineField[0].length;
213 
214         double squareWidth = canvas.getWidth() / fieldX;
215         double squareHeight = canvas.getHeight() / fieldY;
216 
217         GraphicsContext brush = canvas.getGraphicsContext2D();
218 
219         for (int i = 0; i < fieldX; i++) {
220             for (int j = 0; j < fieldY; j++) {
221                 double startX = i * squareWidth;
222                 double startY = j * squareHeight;
223                 Tile tile = mineField[i][j];
224 
225                 if (tile.tileIsOpen()) {
226                     // Background
227                     brush.setFill(Color.WHITE);
228                     brush.fillRect(startX, startY, squareWidth, squareHeight);
229 
230                     // Desired text
231                     brush.setFill(Color.BLACK);
232                     brush.setTextAlign(TextAlignment.CENTER);
233                     brush.setTextBaseline(VPos.CENTER);
234                     
235 
236                     String symbol = logic.getTileSymbol(tile);
237 
238                     brush.fillText(
239                             symbol,
240                             Math.round(startX) + squareWidth / 2,
241                             Math.round(startY) + squareHeight / 2
242                     );
243                 } else if(tile.isFlagged() ) {
244                     brush.setFill(Color.DARKRED);
245                     brush.fillRect(startX, startY, squareWidth, squareHeight);
246                     
247                 } else {
248                     brush.setFill(Color.DARKSEAGREEN);
249                     brush.fillRect(startX, startY, squareWidth, squareHeight);
250                     brush.setFill(Color.DARKGREEN);
251                     brush.fillRect(startX-1, startY-1, squareWidth-1, squareHeight-1);
252                     
253                     
254                 }
255             }
256         }
257         
258     }

261     private void handleMouseClick (MouseEvent event, Canvas canvas, MiinaharavaLogic logic) {
262         double mouseX = event.getX();
263         double width = canvas.getWidth();
264         int fieldWidth = logic.minefield.getMinefield().length;
265 
266         double mouseY = event.getY();
267         double height = canvas.getWidth();
268         int fieldHeight = logic.minefield.getMinefield()[0].length;
269 
270         int x = locationInGrid(mouseX, width, fieldWidth);
271         int y = locationInGrid(mouseY, height, fieldHeight);
272 
273         if ((event.getButton().equals(MouseButton.PRIMARY))) {
274                 
275             logic.sweep(x, y);
276         }
277         
278         if ((event.getButton().equals(MouseButton.SECONDARY))) {
279             
280             logic.setFlag(x, y);
281         }
282        
283         
284         
285     }
